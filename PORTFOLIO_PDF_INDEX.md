```tsx
import { useRef } from 'react';

export default function Portfolio() {
  const ref = useRef<HTMLDivElement>(null);
  return (

        name="John Smith"
        title="Tour Guide"
        sections={[

        ]}
      />

        Download PDF
      </button>
    </>
  );
}
```
